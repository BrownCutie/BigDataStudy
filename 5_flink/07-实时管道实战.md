## 四、Flink 作业开发

### 4.1 完整的实时管道作业

```python
# demo_realtime_pipeline.py
"""
数据分析项目完整实时数据处理管道

功能：
1. 实时 UV/PV 统计（每分钟）
2. 热门文章 Top 10（每 5 分钟滑动窗口）
3. 实时搜索词统计（每分钟）
4. 每篇文章的实时浏览量/评论数/点赞数
"""
import json
from pyflink.datastream import StreamExecutionEnvironment, WatermarkStrategy, Duration
from pyflink.datastream.connectors import KafkaSource, KafkaSink, KafkaRecordSerializationSchema
from pyflink.datastream.functions import KeyedProcessFunction, ProcessWindowFunction
from pyflink.datastream.state import ValueStateDescriptor, MapStateDescriptor
from pyflink.common.serialization import SimpleStringSchema


def create_kafka_source(env, topic, group_id):
    """创建 Kafka Source"""
    return KafkaSource.builder() \
        .set_bootstrap_servers("localhost:9092") \
        .set_topics(topic) \
        .set_group_id(group_id) \
        .set_value_deserialization_schema(SimpleStringSchema()) \
        .build()


def create_kafka_sink(topic):
    """创建 Kafka Sink"""
    return KafkaSink.builder() \
        .set_bootstrap_servers("localhost:9092") \
        .set_record_serializer(
            KafkaRecordSerializationSchema.builder()
                .set_topic(topic)
                .set_value_serialization_schema(SimpleStringSchema())
                .build()
        ) \
        .build()


def main():
    env = StreamExecutionEnvironment.get_execution_environment()

    # 启用 Checkpoint（30 秒间隔）
    env.enable_checkpointing(30000)

    # 设置并行度
    env.set_parallelism(2)

    # ─────────────────────────────────────────────
    # 创建 Kafka Source
    # ─────────────────────────────────────────────
    kafka_source = create_kafka_source(env, "blog-events", "flink-demo-pipeline")

    # Watermark 策略：允许 10 秒延迟
    watermark_strategy = WatermarkStrategy \
        .for_bounded_out_of_orderness(Duration.of_seconds(10)) \
        .with_timestamp_assigner(
            lambda event, ts: json.loads(event).get("timestamp", ts)
        )

    # 从 Kafka 读取并解析
    raw_stream = env.from_source(
        kafka_source,
        watermark_strategy=watermark_strategy,
        source_name="Kafka-BlogEvents"
    )

    # 解析 JSON
    events = raw_stream.map(lambda v: json.loads(v))

    # ─────────────────────────────────────────────
    # 指标 1：实时 UV/PV（每分钟）
    # ─────────────────────────────────────────────
    view_events = events.filter(lambda e: e.get("type") == "view")

    # PV：总浏览次数
    from pyflink.datastream import Window
    pv_stream = view_events \
        .map(lambda e: ("PV", 1)) \
        .key_by(lambda x: x[0]) \
        .window(TumblingEventTimeWindows.of(Duration.of_minutes(1))) \
        .sum(1)

    # UV：独立访客数（简化版，使用 Set 去重）
    uv_stream = view_events \
        .map(lambda e: ("UV", e["userId"])) \
        .key_by(lambda x: x[0]) \
        .window(TumblingEventTimeWindows.of(Duration.of_minutes(1))) \
        .process(UVCounter())

    pv_stream.print("PV:")
    uv_stream.print("UV:")

    # ─────────────────────────────────────────────
    # 指标 2：每篇文章的实时统计
    # ─────────────────────────────────────────────
    article_stats = events \
        .map(lambda e: (e.get("postId", 0), e["type"])) \
        .key_by(lambda x: x[0]) \
        .process(ArticleStatsProcess())

    article_stats.print("文章统计:")

    # ─────────────────────────────────────────────
    # 指标 3：实时搜索词统计
    # ─────────────────────────────────────────────
    search_events = events.filter(lambda e: e.get("type") == "search")

    search_stats = search_events \
        .map(lambda e: (e.get("content", ""), 1)) \
        .key_by(lambda x: x[0]) \
        .window(TumblingEventTimeWindows.of(Duration.of_minutes(1))) \
        .sum(1)

    search_stats.print("搜索词统计:")

    # 执行作业
    env.execute("数据平台实时数据管道")


class UVCounter(ProcessWindowFunction):
    """计算 UV（独立访客数）"""

    def process(self, key, context, elements):
        # 用 Set 去重
        unique_users = set(e[1] for e in elements)
        yield (key, len(unique_users))


class ArticleStatsProcess(KeyedProcessFunction):
    """统计每篇文章的浏览量、评论数、点赞数"""

    def __init__(self):
        self.stats_state = None

    def open(self, parameters):
        descriptor = MapStateDescriptor(
            "article-stats",
            str,   # key: event type
            int    # value: count
        )
        self.stats_state = self.get_runtime_context().get_map_state(descriptor)

    def process_element(self, value, ctx):
        post_id, event_type = value

        # 读取当前计数
        current = self.stats_state.get(event_type)
        if current is None:
            current = 0

        # 更新计数
        self.stats_state.put(event_type, current + 1)

        # 输出当前统计
        views = self.stats_state.get("view") or 0
        comments = self.stats_state.get("comment") or 0
        likes = self.stats_state.get("like") or 0

        yield (post_id, views, comments, likes)


if __name__ == "__main__":
    main()
```

## 六、使用 Flink SQL 版本

如果更倾向用 SQL，这里是纯 SQL 版本：

```python
# demo_sql_realtime_pipeline.py
"""
数据分析项目实时管道 - Flink SQL 版本
"""
from pyflink.table import EnvironmentSettings, TableEnvironment


def main():
    env_settings = EnvironmentSettings.in_streaming_mode()
    table_env = TableEnvironment.create(env_settings)

    # 配置 Checkpoint
    table_env.execute_sql("SET 'execution.checkpointing.interval' = '30s'")

    # 创建 Kafka 源表
    table_env.execute_sql("""
        CREATE TABLE demo_events (
            type STRING,
            postId INT,
            userId STRING,
            content STRING,
            event_time TIMESTAMP(3),
            WATERMARK FOR event_time AS event_time - INTERVAL '10' SECOND
        ) WITH (
            'connector' = 'kafka',
            'topic' = 'blog-events',
            'properties.bootstrap.servers' = 'localhost:9092',
            'properties.group.id' = 'flink-sql-pipeline',
            'format' = 'json',
            'scan.startup.mode' = 'latest-offset'
        )
    """)

    # 创建 Kafka 输出表
    table_env.execute_sql("""
        CREATE TABLE demo_stats_output (
            metric_name STRING,
            metric_key STRING,
            metric_value BIGINT,
            window_start TIMESTAMP(3),
            window_end TIMESTAMP(3)
        ) WITH (
            'connector' = 'kafka',
            'topic' = 'demo-stats-output',
            'properties.bootstrap.servers' = 'localhost:9092',
            'format' = 'json'
        )
    """)

    # 指标 1：每分钟 PV
    table_env.execute_sql("""
        INSERT INTO demo_stats_output
        SELECT
            'PV' AS metric_name,
            'all' AS metric_key,
            COUNT(*) AS metric_value,
            TUMBLE_START(event_time, INTERVAL '1' MINUTE) AS window_start,
            TUMBLE_END(event_time, INTERVAL '1' MINUTE) AS window_end
        FROM demo_events
        WHERE type = 'view'
        GROUP BY TUMBLE(event_time, INTERVAL '1' MINUTE)
    """)

    # 指标 2：每分钟 UV
    table_env.execute_sql("""
        INSERT INTO demo_stats_output
        SELECT
            'UV' AS metric_name,
            'all' AS metric_key,
            COUNT(DISTINCT userId) AS metric_value,
            TUMBLE_START(event_time, INTERVAL '1' MINUTE) AS window_start,
            TUMBLE_END(event_time, INTERVAL '1' MINUTE) AS window_end
        FROM demo_events
        WHERE type = 'view'
        GROUP BY TUMBLE(event_time, INTERVAL '1' MINUTE)
    """)


if __name__ == "__main__":
    main()
```

## 九、面试题

### Q1：如何保证实时管道的数据不丢失？

**参考答案：**

保证数据不丢失需要从三个层面考虑：

1. **Source 端**：使用 Kafka 作为数据源，Kafka 本身有数据持久化能力。Flink 的 Kafka Consumer 将消费偏移量保存在 Checkpoint 中，故障恢复后从上次的位置继续消费。

2. **Flink 内部**：启用 Checkpoint 机制，定期保存算子状态。配置合理的 Checkpoint 间隔（如 30 秒）和重启策略（如固定延迟重启 3 次）。

3. **Sink 端**：使用支持事务的 Sink（如 Kafka 的事务机制、Elasticsearch 的幂等写入），配合 Flink 的两阶段提交实现精确一次语义。

### Q2：如何处理实时管道中的数据倾斜？

**参考答案：**

数据倾斜的典型表现：某些 Task 处理速度远慢于其他 Task，导致整个作业延迟。

解决方案：
1. **两阶段聚合（Local-Global）**：先在局部做预聚合，再全局聚合，减少数据传输
2. **增加并行度**：让数据更均匀分布到多个 Task
3. **重新设计 Key**：如果某个 Key 数据量特别大，可以考虑对 Key 做随机前缀打散
4. **使用 Flink SQL**：SQL 引擎会自动做一些优化

### Q3：如何监控 Flink 作业的健康状态？

**参考答案：**

1. **Web UI**：查看作业状态、反压指标、Checkpoint 耗时、吞吐量
2. **Flink Metrics**：通过 Metrics 系统暴露指标，接入 Prometheus + Grafana
3. **日志告警**：监控 Flink 日志中的 ERROR 和 WARN，配置告警规则
4. **Checkpoint 监控**：Checkpoint 失败率、耗时、大小等指标
5. **业务指标**：监控数据处理的延迟和吞吐量是否符合预期

本篇我们完成了数据分析项目的完整实时管道，关键要点：

1. **管道架构**：Kafka（数据源）→ Flink（实时计算）→ ES/Kafka（存储输出）
2. **多指标并行**：UV/PV、热门文章、搜索词统计同时运行
3. **部署运行**：`flink run -py` 提交，Web UI 监控
4. **故障恢复**：Checkpoint + Savepoint 保证数据不丢失
5. **SQL vs DataStream**：简单聚合用 SQL，复杂逻辑用 DataStream
