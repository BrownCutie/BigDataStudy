### 1.2 关键配置

```xml
<!-- hive-site.xml 中的 HiveServer2 配置 -->

<!-- 绑定地址：0.0.0.0 允许远程连接 -->
<property>
    <name>hive.server2.thrift.bind.host</name>
    <value>0.0.0.0</value>
</property>

<!-- Thrift 端口 -->
<property>
    <name>hive.server2.thrift.port</name>
    <value>10000</value>
</property>

<!-- Web UI 端口 -->
<property>
    <name>hive.server2.webui.port</name>
    <value>10002</value>
</property>

<!-- 最大并发连接数 -->
<property>
    <name>hive.server2.max.sessions</name>
    <value>100</value>
</property>

<!-- 每个会话允许的最大并发操作 -->
<property>
    <name>hive.server2.thrift.max.message.size</name>
    <value>104857600</value>  <!-- 100MB -->
</property>

<!-- 执行超时时间（毫秒），0 为不限制 -->
<property>
    <name>hive.server2.long.running.check.interval</name>
    <value>60000</value>
</property>

<!-- 操作日志开关 -->
<property>
    <name>hive.server2.logging.operation.enabled</name>
    <value>true</value>
</property>

<!-- 操作日志级别 -->
<property>
    <name>hive.server2.logging.operation.log.location</name>
    <value>/opt/hive/4.0.1/logs/operation_logs</value>
</property>
```

### 1.3 用户认证和权限控制

```xml
<!-- 认证方式配置 -->

```

```bash

!connect jdbc:hive2://localhost:10000 browncutie
```

### 2.2 常用命令

```bash

    -f /path/to/query.hql \

beeline -u "jdbc:hive2://localhost:10000" \
    --hivevar dt='2026-04-15' \
    -e "SELECT * FROM dwd.dwd_access_detail WHERE dt = '${hivevar:dt}' LIMIT 10"
```

## 三、动态分区

### 3.1 静态分区 vs 动态分区

```
┌──────────────────────────────────────────────────────────────────┐
│              静态分区 vs 动态分区                                 │
│                                                                  │
│  静态分区：                                                      │
│  INSERT INTO TABLE t PARTITION (dt = '2026-04-15')               │
│  SELECT ... FROM source;                                         │
│  - 分区值在 SQL 中写死                                           │
│  - 每个分区需要一条 INSERT 语句                                  │
│  - 适合：已知分区值的场景                                        │
│                                                                  │
│  动态分区：                                                      │
│  INSERT INTO TABLE t PARTITION (dt)                              │
│  SELECT ..., dt_col FROM source;                                 │
│  - 分区值来自 SELECT 中的字段                                    │
│  - 一条 INSERT 可以写入多个分区                                  │
│  - 适合：分区值不确定或数量很多的场景                            │
│                                                                  │
└──────────────────────────────────────────────────────────────────┘
```

### 3.2 动态分区配置

```sql
-- 开启动态分区（默认 true）
SET hive.exec.dynamic.partition = true;

-- 动态分区模式：
-- strict：至少有一个静态分区列（默认，防止误操作）
-- nonstrict：允许全部列为动态分区
SET hive.exec.dynamic.partition.mode = nonstrict;

-- 最大动态分区数（防止生成过多小文件）
SET hive.exec.max.dynamic.partitions = 1000;

-- 每个 MapReduce 任务的 最大动态分区数
SET hive.exec.max.dynamic.partitions.pernode = 100;
```

### 3.3 动态分区实战

```sql
-- 场景：从 ODS 原始数据写入 DWD，按日期自动分区
-- ODS 数据中包含日期字段 created_at，需要自动创建分区

-- 创建目标分区表
CREATE TABLE dwd.dwd_article_dynamic (
    article_id      BIGINT,
    title           STRING,
    author_id       BIGINT,
    view_count      BIGINT
)
PARTITIONED BY (dt STRING)
STORED AS ORC
TBLPROPERTIES ('orc.compress' = 'SNAPPY');

-- 动态分区写入
SET hive.exec.dynamic.partition = true;
SET hive.exec.dynamic.partition.mode = nonstrict;

INSERT OVERWRITE TABLE dwd.dwd_article_dynamic PARTITION (dt)
SELECT
    id              AS article_id,
    title           AS title,
    author_id       AS author_id,
    view_count      AS view_count,
    -- 动态分区字段必须是 SELECT 的最后一个字段
    SUBSTR(created_at, 1, 10)  AS dt
FROM ods.ods_demo_article
WHERE status = 1;

-- 验证分区
SHOW PARTITIONS dwd.dwd_article_dynamic;
-- 输出：
-- dt=2026-04-10
-- dt=2026-04-11
-- dt=2026-04-12
-- dt=2026-04-13
-- dt=2026-04-14
-- dt=2026-04-15
```

### 3.4 动态分区的注意事项

```
┌──────────────────────────────────────────────────────────────────┐
│              动态分区注意事项                                     │
│                                                                  │
│  1. 小文件问题：                                                │
│     每个分区至少生成一个文件，分区过多会导致大量小文件           │
│     解决：合理控制分区粒度（按天而非按小时）                    │
│                                                                  │
│  2. MapReduce 内存：                                            │
│     动态分区需要在内存中维护分区信息                             │
│     分区过多可能 OOM                                            │
│     解决：调大 hive.exec.max.dynamic.partitions                 │
│                                                                  │
│  3. SELECT 顺序：                                               │
│     分区字段必须放在 SELECT 的最后                              │
│     且与 PARTITION 子句中的顺序一致                             │
│                                                                  │
│  4. strict 模式：                                               │
│     默认 strict 模式要求至少一个静态分区列                      │
│     防止全表扫描后生成过多分区                                  │
│     确认无误后可以改为 nonstrict                                 │
│                                                                  │
└──────────────────────────────────────────────────────────────────┘
```

## 四、视图（VIEW）和物化视图

### 4.1 视图

视图是虚拟表，不存储数据，只保存查询逻辑。每次查询视图时，Hive 会将视图定义展开为实际查询：

```sql
-- 创建视图
CREATE VIEW v_article_hot AS
SELECT
    article_id,
    title,
    author_name,
    view_count,
    CASE
        WHEN view_count >= 5000 THEN '爆款'
        WHEN view_count >= 3000 THEN '热门'
        WHEN view_count >= 1000 THEN '优质'
        ELSE '普通'
    END AS popularity
FROM dwd.dwd_article_publish
WHERE dt = '2026-04-15'
  AND view_count > 0;

-- 使用视图
SELECT * FROM v_article_hot WHERE popularity = '爆款';

-- 查看视图定义
SHOW CREATE TABLE v_article_hot;
DESC FORMATTED v_article_hot;

-- 删除视图
DROP VIEW IF EXISTS v_article_hot;
```

### 4.2 视图的原理

```
┌──────────────────────────────────────────────────────────────────┐
│                    视图工作原理                                   │
│                                                                  │
│  用户查询：                                                      │
│  SELECT * FROM v_article_hot WHERE popularity = '爆款';          │
│       │                                                          │
│       │ Hive 展开视图定义                                       │
│       ▼                                                          │
│  实际执行：                                                      │
│  SELECT * FROM (                                                │
│      SELECT article_id, title, author_name, view_count,         │
│             CASE WHEN view_count >= 5000 THEN '爆款' ... END    │
│      FROM dwd.dwd_article_publish                                │
│      WHERE dt = '2026-04-15' AND view_count > 0                 │
│  ) t WHERE popularity = '爆款';                                  │
│       │                                                          │
│       │ 优化器进一步优化                                        │
│       ▼                                                          │
│  → 谓词下推：view_count >= 5000 推到内层查询                    │
│  → 分区裁剪：dt = '2026-04-15'                                   │
│                                                                  │
│  注意：视图只是"语法糖"，不存储数据，不提升查询性能              │
│  每次查询都会重新执行底层查询                                    │
│                                                                  │
└──────────────────────────────────────────────────────────────────┘
```

### 4.3 物化视图

物化视图（Materialized View）与普通视图不同，它会实际存储查询结果，从而避免重复计算：

```
┌──────────────────────────────────────────────────────────────────┐
│              普通视图 vs 物化视图                                 │
│                                                                  │
│  普通视图 (VIEW)：                                               │
│  ┌──────────┐    ┌──────────┐    ┌──────────┐                  │
│  │ 查询请求 │ →  │ 视图定义 │ →  │ 实际查询 │ → 底层表          │
│  └──────────┘    │ (存储SQL)│    │ (展开SQL)│                  │
│                  └──────────┘    └──────────┘                  │
│  不存数据，每次都重新执行                                       │
│                                                                  │
│  物化视图 (MATERIALIZED VIEW)：                                  │
│  ┌──────────┐    ┌──────────┐    ┌──────────┐                  │
│  │ 查询请求 │ →  │ 物化视图 │ →  │ 直接读取 │ → 预计算结果      │
│  └──────────┘    │ (存数据) │    │ (极快)   │                  │
│                  └──────────┘    └──────────┘                  │
│  存储预计算结果，查询极快                                       │
│  需要定期刷新（REBUILD）                                        │
│                                                                  │
└──────────────────────────────────────────────────────────────────┘
```

```sql
-- 创建物化视图
CREATE MATERIALIZED VIEW mv_article_daily_stats
DISABLE REWRITE
AS
SELECT
    article_id,
    COUNT(*) AS pv_count,
    COUNT(DISTINCT user_id) AS uv_count,
    AVG(duration_sec) AS avg_duration
FROM dwd.dwd_access_detail
GROUP BY article_id;

-- 刷新物化视图（重新执行底层查询）
ALTER MATERIALIZED VIEW mv_article_daily_stats REBUILD;

-- 查询物化视图
SELECT * FROM mv_article_daily_stats ORDER BY pv_count DESC LIMIT 10;

-- 删除物化视图
DROP MATERIALIZED VIEW IF EXISTS mv_article_daily_stats;

-- 注意：Hive 4.x 的物化视图功能仍在演进中
-- 生产环境建议使用 DolphinScheduler 定时刷新普通表来替代
```

## 五、Hive Script（存储过程）

### 5.1 Hive Script 简介

Hive 支持通过 SQL 脚本实现类似存储过程的功能：

```bash
# 创建 SQL 脚本文件
cat > /opt/hive/scripts/daily_etl.hql << 'EOF'
-- 数据平台每日 ETL 脚本
-- 使用方法：hive -f daily_etl.hql --hivevar run_date=2026-04-15

SET hive.exec.dynamic.partition = true;
SET hive.exec.dynamic.partition.mode = nonstrict;
SET hive.exec.compress.output = true;
SET mapreduce.output.fileoutputformat.compress.codec = org.apache.hadoop.io.compress.SnappyCodec;

-- Step 1: ODS → DWD 清洗
INSERT OVERWRITE TABLE dwd.dwd_article_publish PARTITION (dt)
SELECT
    id, title, author_id, category_id,
    LENGTH(content) AS word_count,
    CAST(created_at AS DATE) AS publish_date,
    title AS tags,
    SUBSTR(created_at, 1, 10) AS dt
FROM ods.ods_demo_article
WHERE dt = '${hivevar:run_date}'
  AND status = 1
  AND content IS NOT NULL;

-- Step 2: DWD → DWS 汇总
INSERT OVERWRITE TABLE dws.dws_article_daily_stats PARTITION (dt)
SELECT
    a.article_id,
    a.title,
    a.author_name,
    COUNT(*) AS pv_count,
    COUNT(DISTINCT c.user_id) AS uv_count,
    ROUND(AVG(c.duration_sec), 2) AS avg_duration
FROM dwd.dwd_article_publish a
LEFT JOIN dwd.dwd_access_detail c
    ON a.article_id = c.article_id AND c.dt = '${hivevar:run_date}'
WHERE a.dt = '${hivevar:run_date}'
GROUP BY a.article_id, a.title, a.author_name;

-- Step 3: DWS → ADS 报表
INSERT OVERWRITE TABLE ads.ads_article_ranking
SELECT
    article_id,
    title,
    author_name,
    pv_count,
    uv_count,
    (pv_count * 0.3 + uv_count * 10.0) AS rank_score,
    ROW_NUMBER() OVER (ORDER BY (pv_count * 0.3 + uv_count * 10.0) DESC) AS rank_pos
FROM dws.dws_article_daily_stats
WHERE dt = '${hivevar:run_date}';

-- 完成
SELECT 'ETL completed for date: ${hivevar:run_date}' AS status;
EOF

# 执行脚本
hive -f /opt/hive/scripts/daily_etl.hql --hivevar run_date=2026-04-15

# 使用 Beeline 执行
beeline -u "jdbc:hive2://localhost:10000" \
    -f /opt/hive/scripts/daily_etl.hql \
    --hivevar run_date=2026-04-15
```

### 5.2 脚本调度

```bash
# 在 DolphinScheduler 中调度 Hive 脚本
# 创建工作流，每日凌晨 2:00 执行

# 也可以使用 crontab 调度（简单场景）
# crontab -e
# 0 2 * * * /opt/hive/scripts/run_daily_etl.sh

cat > /opt/hive/scripts/run_daily_etl.sh << 'EOF'
#!/bin/bash
# 每日 ETL 执行脚本

RUN_DATE=$(date -d "yesterday" +%Y-%m-%d)
LOG_FILE="/opt/hive/logs/etl_${RUN_DATE}.log"

echo "=== ETL Start: $(date) ===" > $LOG_FILE
echo "Processing date: $RUN_DATE" >> $LOG_FILE

beeline -u "jdbc:hive2://localhost:10000" \
    -f /opt/hive/scripts/daily_etl.hql \
    --hivevar run_date=$RUN_DATE \
    >> $LOG_FILE 2>&1

if [ $? -eq 0 ]; then
    echo "=== ETL Success ===" >> $LOG_FILE
else
    echo "=== ETL Failed ===" >> $LOG_FILE
    # 发送告警（邮件/钉钉/企业社交平台）
fi

echo "=== ETL End: $(date) ===" >> $LOG_FILE
EOF

chmod +x /opt/hive/scripts/run_daily_etl.sh
```

## 六、Hive 与 Spark 的区别

### 6.1 核心区别

```
┌──────────────────────────────────────────────────────────────────┐
│                Hive vs Spark 对比                                 │
│                                                                  │
│                    Hive                    Spark                  │
│  ┌────────────────────────┐  ┌────────────────────────┐        │
│  │ 定位: 批处理数仓工具   │  │ 定位: 通用计算引擎     │        │
│  │                        │  │                        │        │
│  │ 查询语言: HQL (SQL)    │  │ 支持: SQL / Scala /    │        │
│  │                        │  │  Python / Java / R     │        │
│  │ 执行引擎:              │  │                        │        │
│  │  MapReduce (慢)        │  │ 执行引擎:              │        │
│  │  Tez (中)              │  │  自有 DAG 引擎 (快)    │        │
│  │  Spark (快)            │  │  基于内存计算          │        │
│  │                        │  │                        │        │
│  │ 延迟: 分钟级            │  │ 延迟: 秒级              │        │
│  │                        │  │                        │        │
```

### 7.2 MERGE INTO 示例

```sql
-- MERGE INTO 是 Hive 4.x 新增的 SQL 标准语法
-- 类似于 "UPSERT"（存在则更新，不存在则插入）

-- 目标表
CREATE TABLE dwd.dwd_user_profile_merge (
    user_id         BIGINT,
    username        STRING,
    nickname        STRING,
    total_articles  INT,
    total_views     BIGINT
)
CLUSTERED BY (user_id) INTO 4 BUCKETS
STORED AS ORC
TBLPROPERTIES ('transactional' = 'true');

-- 源数据
CREATE TEMPORARY TABLE tmp_user_updates (
    user_id         BIGINT,
    username        STRING,
    nickname        STRING,
    total_articles  INT,
    total_views     BIGINT
);

INSERT INTO tmp_user_updates VALUES
    (2001, 'zhangsan', '张三', 8, 2500),   -- 已存在，更新
    (2004, 'lisi',     '李四', 2, 800),     -- 不存在，插入
    (2005, 'wangwu',   '王五', 3, 1200);    -- 不存在，插入

-- MERGE INTO 操作
MERGE INTO dwd.dwd_user_profile_merge AS target
USING tmp_user_updates AS source
ON target.user_id = source.user_id
WHEN MATCHED THEN
    UPDATE SET
        target.username = source.username,
        target.nickname = source.nickname,
        target.total_articles = source.total_articles,
        target.total_views = source.total_views
WHEN NOT MATCHED THEN
    INSERT (user_id, username, nickname, total_articles, total_views)
    VALUES (source.user_id, source.username, source.nickname,
            source.total_articles, source.total_views);

-- 验证
SELECT * FROM dwd.dwd_user_profile_merge ORDER BY user_id;
```

### 7.3 数据分析项目中的 Hive 4.x 特性应用

```sql
-- 1. 使用 VALUES 子句直接插入多行（4.x 增强）
INSERT INTO ads.ads_article_ranking
VALUES
    (1001, 'Hive教程', '张三', '大数据', 3500, 200, 3050.0, 1),
    (1002, 'Spark教程', '张三', '大数据', 5200, 350, 5110.0, 2);

-- 2. 使用改进的窗口函数
SELECT
    article_id,
    title,
    view_count,
    -- 百分位排名
    PERCENT_RANK() OVER (ORDER BY view_count DESC) AS pct_rank,
    -- 累计分布
    CUME_DIST() OVER (ORDER BY view_count DESC) AS cume_dist,
    -- N 个桶
    NTILE(4) OVER (ORDER BY view_count DESC) AS quartile
FROM dwd.dwd_article_clean;

-- 3. 使用物化视图加速查询
-- 对于每天执行的热门查询，可以创建物化视图避免重复计算
CREATE MATERIALIZED VIEW mv_daily_top_articles
DISABLE REWRITE
AS
SELECT
    a.article_id,
    a.title,
    a.author_name,
    COUNT(*) AS pv_count,
    COUNT(DISTINCT c.user_id) AS uv_count
FROM dwd.dwd_article_publish a
LEFT JOIN dwd.dwd_access_detail c
    ON a.article_id = c.article_id AND c.dt = a.dt
WHERE a.dt = CURRENT_DATE - 1
GROUP BY a.article_id, a.title, a.author_name
ORDER BY pv_count DESC
LIMIT 20;
```
