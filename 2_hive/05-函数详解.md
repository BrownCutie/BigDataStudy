## 一、数学函数

### 1.1 常用数学函数

```sql
-- 创建测试数据
CREATE TEMPORARY TABLE tmp_math_test (val DOUBLE);
INSERT INTO tmp_math_test VALUES (3.14159), (-7.8), (256.6), (0), (42.5);

-- ROUND：四舍五入（第二个参数为保留小数位数）
SELECT val, ROUND(val)            AS r0,        -- 四舍五入到整数
             ROUND(val, 1)        AS r1,        -- 保留1位小数
             ROUND(val, 2)        AS r2         -- 保留2位小数
FROM tmp_math_test;
-- 3.14159  →  3    3.1    3.14
-- -7.8     → -8   -7.8   -7.80
-- 42.5     → 43   42.5   42.50

-- FLOOR：向下取整（不大于该值的最大整数）
SELECT val, FLOOR(val) FROM tmp_math_test;
-- 3.14159  →  3
-- -7.8     → -8
-- 256.6    → 256

-- CEIL / CEILING：向上取整（不小于该值的最小整数）
SELECT val, CEIL(val) FROM tmp_math_test;
-- 3.14159  →  4
-- -7.8     → -7
-- 256.6    → 257

-- ABS：绝对值
SELECT val, ABS(val) FROM tmp_math_test;
-- -7.8  →  7.8

-- POW / POWER：幂运算
SELECT POW(2, 10);         -- 1024.0
SELECT POWER(3, 3);        -- 27.0

-- SQRT：平方根
SELECT SQRT(144);          -- 12.0

-- MOD：取模
SELECT MOD(10, 3);         -- 1
SELECT 10 % 3;             -- 1（等价写法）

-- RAND：返回 0-1 之间的随机数
SELECT RAND();             -- 0.7342...（每次不同）
SELECT RAND(42);           -- 固定种子，每次结果相同

-- GREATEST / LEAST：返回最大/最小值
SELECT GREATEST(10, 20, 5, 30);   -- 30
SELECT LEAST(10, 20, 5, 30);      -- 5
```

### 1.2 数据分析项目数学函数实战

```sql
-- 计算文章平均浏览量、中位数等统计指标
SELECT
    COUNT(*)                            AS total_articles,
    ROUND(AVG(view_count), 2)           AS avg_views,
    ROUND(MAX(view_count) - MIN(view_count), 2) AS views_range,
    GREATEST(MAX(view_count), 0)        AS max_views
FROM dwd.dwd_article_clean;

-- 使用 RAND 实现随机抽样（用于数据探索）
SELECT * FROM dwd.dwd_article_clean
WHERE RAND() < 0.3          -- 随机抽取约 30% 的数据
LIMIT 10;
```

## 二、字符串函数

### 2.1 常用字符串函数

```sql
-- 创建测试数据
CREATE TEMPORARY TABLE tmp_str_test (s STRING);
INSERT INTO tmp_str_test VALUES
    ('Hello, Hive World!'),
    ('  前后有空格  '),
    ('2026-04-15'),
    ('hive-hadoop-spark');

-- SUBSTR / SUBSTRING：截取子串（起始位置从1开始）
SELECT s, SUBSTR(s, 1, 5)            AS s1,     -- 从第1位截取5个字符
            SUBSTR(s, -6)             AS s2      -- 从倒数第6位截到末尾
FROM tmp_str_test;
-- 'Hello, Hive World!'  →  'Hello'    'World!'
-- '2026-04-15'          →  '2026-'    '04-15'

-- CONCAT：字符串拼接（任一参数为 NULL 则结果为 NULL）
SELECT CONCAT('Hive', '-', '4.0.1');          -- 'Hive-4.0.1'

-- CONCAT_WS：带分隔符的拼接（忽略 NULL 值）
SELECT CONCAT_WS(',', 'Hive', 'Spark', 'Flink');     -- 'Hive,Spark,Flink'
SELECT CONCAT_WS('|', 'a', NULL, 'b');               -- 'a|b'（NULL 被忽略）

-- REPLACE：替换字符串
SELECT REPLACE('Hello World', 'World', 'Hive');     -- 'Hello Hive'

-- UPPER / LOWER：大小写转换
SELECT UPPER('hello'), LOWER('HELLO');              -- 'HELLO'  'hello'

-- TRIM / LTRIM / RTRIM：去除空格
SELECT TRIM('  hello  ');          -- 'hello'
SELECT LTRIM('  hello  ');         -- 'hello  '
SELECT RTRIM('  hello  ');         -- '  hello'

-- LENGTH / CHAR_LENGTH：字符串长度
SELECT LENGTH('Hello 你好');       -- 9（UTF-8 字节数）
SELECT CHAR_LENGTH('Hello 你好');  -- 7（字符数）

-- SPLIT：按分隔符拆分为数组
SELECT SPLIT('hive,hadoop,spark', ',');
-- ["hive","hadoop","spark"]

-- REVERSE：反转字符串
SELECT REVERSE('Hello');            -- 'olleH'

-- REPEAT：重复字符串
SELECT REPEAT('Ha', 3);             -- 'HaHaHa'

-- LPAD / RPAD：左右填充
SELECT LPAD('42', 5, '0');          -- '00042'
SELECT RPAD('Hi', 8, '-');          -- 'Hi------'

-- INSTR：查找子串位置（从1开始，未找到返回0）
SELECT INSTR('Hello World', 'World');    -- 7
SELECT INSTR('Hello World', 'hive');     -- 0
```

## 三、日期函数

### 3.1 常用日期函数

```sql
-- 获取当前日期和时间
SELECT CURRENT_DATE;              -- 2026-04-15
SELECT CURRENT_TIMESTAMP;        -- 2026-04-15 14:30:00.123

-- DATE_FORMAT：日期格式化
SELECT DATE_FORMAT(CURRENT_TIMESTAMP, 'yyyy-MM-dd');           -- 2026-04-15
SELECT DATE_FORMAT(CURRENT_TIMESTAMP, 'yyyy-MM-dd HH:mm:ss');  -- 2026-04-15 14:30:00
SELECT DATE_FORMAT(CURRENT_TIMESTAMP, 'yyyy年MM月dd日');       -- 2026年04月15日

-- DATEDIFF：计算两个日期之间的天数差
SELECT DATEDIFF('2026-04-20', '2026-04-15');   -- 5
SELECT DATEDIFF('2026-04-15', '2026-04-20');   -- -5

-- DATE_ADD / DATE_SUB：日期加减
SELECT DATE_ADD('2026-04-15', 7);              -- 2026-04-22
SELECT DATE_SUB('2026-04-15', 3);              -- 2026-04-12

-- YEAR / MONTH / DAY / HOUR / MINUTE / SECOND：提取日期部分
SELECT YEAR('2026-04-15');        -- 2026
SELECT MONTH('2026-04-15');       -- 4
SELECT DAY('2026-04-15');         -- 15
SELECT HOUR('2026-04-15 14:30');  -- 14

-- DAYOFWEEK：星期几（1=周日，2=周一，...，7=周六）
SELECT DAYOFWEEK('2026-04-15');   -- 4（周三）

-- TO_DATE：将时间戳转为日期
SELECT TO_DATE('2026-04-15 14:30:00');  -- 2026-04-15

-- DATE：等价于 TO_DATE
SELECT DATE('2026-04-15 14:30:00');     -- 2026-04-15

-- FROM_UNIXTIME：Unix 时间戳转日期字符串
SELECT FROM_UNIXTIME(1744708800, 'yyyy-MM-dd');  -- 2026-04-15

-- UNIX_TIMESTAMP：日期字符串转 Unix 时间戳
SELECT UNIX_TIMESTAMP('2026-04-15 00:00:00');    -- 1744708800

-- LAST_DAY：获取当月最后一天
SELECT LAST_DAY('2026-04-15');     -- 2026-04-30

-- TRUNC：日期截断
SELECT TRUNC('2026-04-15', 'MM');  -- 2026-04-01（截断到月初）
SELECT TRUNC('2026-04-15', 'YY');  -- 2026-01-01（截断到年初）
```

### 3.2 数据分析项目日期函数实战

```sql
-- 计算文章发布天数
SELECT
    article_id,
    title,
    publish_date,
    DATEDIFF(CURRENT_DATE, publish_date) AS days_since_publish,
    CASE
        WHEN DATEDIFF(CURRENT_DATE, publish_date) <= 7 THEN '本周'
        WHEN DATEDIFF(CURRENT_DATE, publish_date) <= 30 THEN '本月'
        ELSE '更早'
    END AS publish_period
FROM dwd.dwd_article_clean
ORDER BY publish_date DESC;

-- 生成最近 7 天的日期序列（用于填充报表空缺日期）
SELECT DATE_ADD(CURRENT_DATE, -d) AS date_val
FROM (
    SELECT POW(2, 0) + 0 AS d UNION ALL SELECT 1 UNION ALL SELECT 2
    UNION ALL SELECT 3 UNION ALL SELECT 4 UNION ALL SELECT 5 UNION ALL SELECT 6
) t
ORDER BY date_val;
```

## 四、条件函数

### 4.1 IF 函数

```sql
-- IF(condition, true_value, false_value)
SELECT
    article_id,
    title,

    IF(view_count > 5000, '高热度', '普通') AS heat_level
```

### 4.2 CASE WHEN

```sql
-- 简单 CASE
SELECT
    category_id,
    CASE category_id
        WHEN 1 THEN '大数据'
        WHEN 2 THEN '实时计算'
        WHEN 3 THEN '消息队列'
        WHEN 4 THEN '搜索引擎'
        ELSE '其他'

    article_id,

    view_count,
    CASE
        WHEN view_count >= 6000 THEN '爆款'
        WHEN view_count >= 4000 THEN '热门'
        WHEN view_count >= 2000 THEN '优质'
        WHEN view_count >= 500  THEN '普通'
```

### 4.3 COALESCE 和 NVL

```sql
-- COALESCE：返回第一个非 NULL 值
SELECT
    user_id,
    COALESCE(user_id, -1)          AS user_id_safe,    -- NULL 替换为 -1
    COALESCE(NULL, NULL, 'default')  AS fallback;       -- 返回 'default'

-- NVL：等价于 COALESCE 的两参数版本
SELECT NVL(NULL, '匿名用户');      -- '匿名用户'

-- 数据分析项目实战：处理 NULL 值
SELECT
    article_id,
    COALESCE(author_name, '佚名')     AS display_author,
    COALESCE(view_count, 0)           AS display_views,
    NVL(category_name, '未分类')      AS display_category
FROM dwd.dwd_article_publish
WHERE dt = '2026-04-15';
```

### 4.4 NULLIF

## 五、聚合函数

### 5.1 基本聚合

```sql
-- COUNT：计数
SELECT COUNT(*)                    AS total,          -- 所有行（含 NULL）
       COUNT(article_id)          AS non_null_count, -- 非NULL行
       COUNT(DISTINCT author_id)  AS unique_authors  -- 去重计数
FROM dwd.dwd_article_clean;

-- SUM / AVG / MAX / MIN
SELECT
    SUM(view_count)                     AS total_views,
    AVG(view_count)                     AS avg_views,
    MAX(view_count)                     AS max_views,
    MIN(view_count)                     AS min_views,
    ROUND(AVG(view_count), 2)           AS avg_views_rounded
FROM dwd.dwd_article_clean;

-- GROUP BY 分组聚合
SELECT
    author_id,
    COUNT(*)                AS article_count,
    SUM(view_count)         AS total_views,
    ROUND(AVG(view_count), 2) AS avg_views
FROM dwd.dwd_article_clean
GROUP BY author_id
ORDER BY total_views DESC;

-- HAVING：对分组后的结果过滤（WHERE 过滤行，HAVING 过滤组）
SELECT
    author_id,
    COUNT(*) AS article_count,
    SUM(view_count) AS total_views
FROM dwd.dwd_article_clean
GROUP BY author_id
HAVING COUNT(*) >= 2          -- 只保留文章数 >= 2 的作者
ORDER BY total_views DESC;
```

### 5.2 GROUP BY + CASE WHEN 实现多维度统计

```sql
-- 按浏览量区间统计文章数量
SELECT
    CASE
        WHEN view_count >= 6000 THEN '6000+'
        WHEN view_count >= 4000 THEN '4000-5999'
        WHEN view_count >= 2000 THEN '2000-3999'
        ELSE '0-1999'
    END AS view_range,
    COUNT(*) AS article_count
FROM dwd.dwd_article_clean
GROUP BY
    CASE
        WHEN view_count >= 6000 THEN '6000+'
        WHEN view_count >= 4000 THEN '4000-5999'
        WHEN view_count >= 2000 THEN '2000-3999'
        ELSE '0-1999'
    END
ORDER BY article_count DESC;

-- 多维度 GROUPING SETS（等价于多个 GROUP BY 的 UNION ALL）
SELECT
    author_id,
    category_id,
    COUNT(*)        AS cnt,
    SUM(view_count) AS views
FROM dwd.dwd_article_clean
GROUP BY GROUPING SETS (
    (author_id),          -- 按作者汇总
    (category_id),        -- 按分类汇总
    (author_id, category_id),  -- 按作者+分类汇总
    ()                    -- 全局汇总
);
```

## 六、窗口函数

窗口函数是 Hive 中最强大的分析工具，允许在不改变行数的情况下进行跨行计算。

### 6.1 基本语法

```sql
窗口函数 OVER (
    PARTITION BY col1, col2    -- 分组（类似 GROUP BY，但不合并行）
    ORDER BY col3              -- 排序（定义窗口内的顺序）
    ROWS BETWEEN start AND end -- 窗口范围
)
```

### 6.2 排名函数

```sql
-- 准备测试数据
CREATE TEMPORARY TABLE tmp_rank_test (
    author_id   BIGINT,
    article_id  BIGINT,
    view_count  BIGINT
);
INSERT INTO tmp_rank_test VALUES
    (1001, 101, 6100), (1001, 102, 3500), (1001, 103, 5500),
    (1002, 104, 4800), (1002, 105, 4200),
    (1003, 106, 3900);

-- ROW_NUMBER：连续排名（1,2,3,4...），遇到相同值也继续递增
SELECT
    author_id, article_id, view_count,
    ROW_NUMBER() OVER (PARTITION BY author_id ORDER BY view_count DESC) AS rn
FROM tmp_rank_test;
-- 1001 | 101 | 6100 | 1
-- 1001 | 103 | 5500 | 2
-- 1001 | 102 | 3500 | 3
-- 1002 | 104 | 4800 | 1
-- 1002 | 105 | 4200 | 2

-- RANK：跳跃排名（1,2,2,4...），相同值同排名，后续跳过
SELECT
    author_id, article_id, view_count,
    RANK() OVER (PARTITION BY author_id ORDER BY view_count DESC) AS rnk
FROM tmp_rank_test;

-- DENSE_RANK：密集排名（1,2,2,3...），相同值同排名，后续不跳过
SELECT
    author_id, article_id, view_count,
    DENSE_RANK() OVER (PARTITION BY author_id ORDER BY view_count DESC) AS drnk
FROM tmp_rank_test;

-- 三种排名的区别：
-- 假设分数为：100, 100, 90, 80
-- ROW_NUMBER → 1, 2, 3, 4
-- RANK       → 1, 1, 3, 4
-- DENSE_RANK → 1, 1, 2, 3
```

### 6.3 偏移函数 LEAD / LAG

```sql
-- LAG：获取前面第 N 行的值（向前看）
-- LEAD：获取后面第 N 行的值（向后看）

SELECT
    article_id,
    publish_date,
    view_count,
    LAG(view_count, 1) OVER (ORDER BY publish_date) AS prev_day_views,
    LEAD(view_count, 1) OVER (ORDER BY publish_date) AS next_day_views,
    view_count - LAG(view_count, 1) OVER (ORDER BY publish_date) AS day_over_day_diff
FROM dwd.dwd_article_clean
ORDER BY publish_date;

-- 结果示例：
-- article_id | publish_date | view_count | prev_day_views | next_day_views | day_over_day_diff
-- 101        | 2026-04-10  | 3500       | NULL           | 5200           | NULL
-- 102        | 2026-04-11  | 5200       | 3500          | 4800           | 1700
-- 103        | 2026-04-12  | 4800       | 5200          | 6100           | -400
-- 104        | 2026-04-13  | 6100       | 4800          | 3900           | 1300
```

### 6.4 首尾值函数

```sql
-- FIRST_VALUE：分组内第一个值
-- LAST_VALUE：分组内最后一个值

SELECT
    author_id,
    article_id,
    view_count,
    FIRST_VALUE(title) OVER (
        PARTITION BY author_id ORDER BY publish_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS first_article_title,
    LAST_VALUE(title) OVER (
        PARTITION BY author_id ORDER BY publish_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS last_article_title
FROM dwd.dwd_article_clean;
```

### 6.5 累计聚合

```sql
-- 累计求和
SELECT
    publish_date,
    view_count,
    SUM(view_count) OVER (
        ORDER BY publish_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_views
FROM dwd.dwd_article_clean
ORDER BY publish_date;

-- 窗口范围说明：
-- ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW  → 从第一行到当前行（累计）
-- ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING          → 前一行到后一行（滑动窗口）
-- ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING → 全部行
```

### 6.6 数据分析项目窗口函数实战：文章排行榜

```sql
-- 每个作者的 Top-N 文章（使用 ROW_NUMBER + 过滤）
WITH ranked_articles AS (
    SELECT
        article_id,
        title,
        author_id,
        author_name,
        view_count,
        ROW_NUMBER() OVER (
            PARTITION BY author_id
            ORDER BY view_count DESC
        ) AS author_rank
    FROM dwd.dwd_article_publish
    WHERE dt = '2026-04-15'
)
SELECT * FROM ranked_articles WHERE author_rank <= 3;

-- 全站文章排行榜
SELECT
    article_id,
    title,
    author_name,
    category_name,
    view_count,
    RANK() OVER (ORDER BY view_count DESC) AS global_rank
FROM dwd.dwd_article_publish
WHERE dt = '2026-04-15'
ORDER BY global_rank;
```

## 七、表生成函数

### 7.1 EXPLODE

`EXPLODE` 将数组或 MAP 展开为多行，是处理嵌套数据的核心函数：

```sql
-- 创建测试数据
CREATE TEMPORARY TABLE tmp_explode_test (
    article_id  BIGINT,
    title       STRING,
    tags        ARRAY<STRING>
);
INSERT INTO tmp_explode_test VALUES
    (101, 'Hive教程',   ARRAY('Hive', '大数据', '数仓')),
    (102, 'Spark教程',  ARRAY('Spark', '分布式', '内存计算')),
    (103, 'Flink教程',  ARRAY('Flink', '流处理'));

-- EXPLODE 数组：每个元素一行
SELECT
    article_id,
    title,
    tag
FROM tmp_explode_test
LATERAL VIEW EXPLODE(tags) t AS tag;
-- 101 | Hive教程  | Hive
-- 101 | Hive教程  | 大数据
-- 101 | Hive教程  | 数仓
-- 102 | Spark教程 | Spark
-- 102 | Spark教程 | 分布式
-- 102 | Spark教程 | 内存计算
-- 103 | Flink教程 | Flink
-- 103 | Flink教程 | 流处理

-- EXPLODE MAP：每个键值对一行
CREATE TEMPORARY TABLE tmp_map_test (
    article_id  BIGINT,
    props       MAP<STRING, STRING>
);
INSERT INTO tmp_map_test VALUES
    (101, MAP('difficulty', '入门', 'duration', '3小时', 'language', '中文'));

SELECT
    article_id,
    prop_key,
    prop_value
FROM tmp_map_test
LATERAL VIEW EXPLODE(props) t AS prop_key, prop_value;
-- 101 | difficulty | 入门
-- 101 | duration   | 3小时
-- 101 | language   | 中文
```

### 7.2 LATERAL VIEW 详解

`LATERAL VIEW` 将 EXPLODE 的结果与原表的每行进行交叉连接（CROSS JOIN）：

```
┌──────────────────────────────────────────────────────────────────┐
│              LATERAL VIEW 工作原理                               │
│                                                                  │
│  原始表：                LATERAL VIEW EXPLODE(tags):             │
│  ┌────────────────┐      ┌──────────────────────┐               │
│  │ 101 | [a,b,c] │  ×   │ EXPLODE → a, b, c    │               │
│  │ 102 | [d,e]   │      └──────────────────────┘               │
│  └────────────────┘                                             │
│       │                                                          │
│       │ LATERAL VIEW（每行与展开结果做 CROSS JOIN）              │
│       ▼                                                          │
│  ┌──────────────────────┐                                       │
│  │ 101 | a              │                                       │
│  │ 101 | b              │                                       │
│  │ 101 | c              │                                       │
│  │ 102 | d              │                                       │
│  │ 102 | e              │                                       │
│  └──────────────────────┘                                       │
│                                                                  │
│  LATERAL VIEW OUTER：保留 EXPLODE 结果为空的行                  │
│  （普通 LATERAL VIEW 会丢弃空数组/NULL 的行）                   │
│                                                                  │
└──────────────────────────────────────────────────────────────────┘
```

### 7.3 数据分析项目 EXPLODE 实战

```sql
-- 标签统计分析：每个标签的文章数
SELECT
    tag,
    COUNT(DISTINCT article_id) AS article_count
FROM (
    SELECT article_id, tag
    FROM dim.dim_article_ext
    LATERAL VIEW EXPLODE(tags) t AS tag
) tag_expanded
GROUP BY tag
ORDER BY article_count DESC;

-- 多列同时 EXPLODE
SELECT
    article_id,
    tag,
    prop_key,
    prop_value
FROM dim.dim_article_ext
LATERAL VIEW EXPLODE(tags) t1 AS tag
LATERAL VIEW EXPLODE(metadata) t2 AS prop_key, prop_value;
```

### 8.2 编写 Java UDF 示例

```java
// 文件：DemoHeatScoreUDF.java
// 功能：根据浏览量和点赞数计算文章热度评分
// 公式：score = view_count * 0.3 + like_count * 10

package com.demo.udf;

import org.apache.hadoop.hive.ql.exec.UDF;
import org.apache.hadoop.io.DoubleWritable;
import org.apache.hadoop.io.LongWritable;

public class DemoHeatScoreUDF extends UDF {

    /**
     * 计算文章热度评分
     * @param viewCount  浏览量
     * @param likeCount  点赞数
     * @return 热度评分
     */
    public DoubleWritable evaluate(LongWritable viewCount, LongWritable likeCount) {
        if (viewCount == null || likeCount == null) {
            return new DoubleWritable(0.0);
        }
        double score = viewCount.get() * 0.3 + likeCount.get() * 10.0;
        return new DoubleWritable(Math.round(score * 100.0) / 100.0);
    }
}
```

### 8.3 编译和注册 UDF

```bash
# 编译 UDF（假设 Maven 项目已配置）
cd /path/to/udf-project
mvn clean package

# 将 JAR 包上传到 Hive 的 lib 目录或任意路径
cp target/demo-udf-1.0.jar /opt/hive/4.0.1/lib/

# 或者在 Beeline 中临时注册（不需要放入 lib/）
```

```sql
-- 在 Beeline 中注册和使用 UDF

-- 方式1：创建永久函数（推荐）
CREATE FUNCTION demo_heat_score
AS 'com.demo.udf.DemoHeatScoreUDF'
USING JAR 'hdfs:///user/hive/udf/demo-udf-1.0.jar';

-- 方式2：临时函数（仅当前会话有效）
ADD JAR /opt/hive/4.0.1/lib/demo-udf-1.0.jar;
CREATE TEMPORARY FUNCTION demo_heat_score
AS 'com.demo.udf.DemoHeatScoreUDF';

-- 使用 UDF
SELECT
    article_id,
    title,
    view_count,
    like_count,
    demo_heat_score(view_count, like_count) AS heat_score
FROM dwd.dwd_article_clean
ORDER BY heat_score DESC;

-- 查看已注册的函数
SHOW FUNCTIONS LIKE 'demo*';

-- 删除函数
DROP FUNCTION IF EXISTS demo_heat_score;
```
